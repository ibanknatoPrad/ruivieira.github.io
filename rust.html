List folders recursively

Using the glob crate:

use glob::glob;

fn main() {

    for entry in glob("./**/*.md").expect("Failed to read glob pattern") {
        match entry {
            Ok(path) => println!("{:?}", path.display()),
            Err(e) => println!("{:?}", e),
        }
    }
}
